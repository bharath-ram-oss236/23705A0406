import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        long[] memo = new long[100];  // Enough for fib(70)
        
        System.out.println("Fibonacci of 30: " + fibMemorization(30, memo));
        System.out.println("Fibonacci of 50: " + fibMemorization(50, memo));
        System.out.println("Fibonacci of 70: " + fibMemorization(70, memo));
        
        System.out.println("Memoization Array:");
        System.out.println(Arrays.toString(memo));
    }

    // Memoization method for Fibonacci
    public static long fibMemorization(long n, long[] memo) {
        if (n == 0 || n == 1) {
            return n;
        }

        if (memo[(int) n] != 0) {
            return memo[(int) n];
        }

        long val = fibMemorization(n - 1, memo) + fibMemorization(n - 2, memo);
        memo[(int) n] = val;

        return val;
    }
}
